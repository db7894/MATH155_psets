\begin{enumerate}
	\item Find the autocovariance function for $W_t = \nabla Y_t = Y_t - Y_{t-1}$ in terms of $\phi$ and $\sigma_e^2$.
	\item In particular, show that $Var(W_T) = 2\sigma_e^2/(1+\phi)$.
\end{enumerate}
